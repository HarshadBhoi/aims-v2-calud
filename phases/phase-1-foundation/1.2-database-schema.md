# 1.2 — Database Schema & ORM

**Parent**: [Phase 1 — Foundation](OVERVIEW.md)
**Estimated effort**: 3-4 days

---

## Objective

Design and implement the core PostgreSQL database schema with Prisma ORM. This schema must support multi-tenancy, multi-standard auditing, and extensibility.

### 1.2.2 — Core Tables (Tenant & User)

```prisma
model Tenant {
  id            String   @id @default(cuid())
  name          String
  slug          String   @unique    // e.g., "city-of-austin-audit"
  logoUrl       String?
  settings      Json     @default("{}")  // Tenant-level config
  subscription  String   @default("free") // free, professional, enterprise
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt

  users         User[]
  engagements   Engagement[]
  standards     TenantStandard[]
  // ... all tenant-scoped relations
}

model User {
  id            String   @id @default(cuid())
  tenantId      String
  tenant        Tenant   @relation(fields: [tenantId], references: [id])
  email         String
  name          String
  passwordHash  String?
  avatarUrl     String?
  role          Role     @default(AUDITOR)
  title         String?          // "Senior Auditor", "Audit Director"
  department    String?
  isActive      Boolean  @default(true)
  lastLoginAt   DateTime?
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt

  @@unique([tenantId, email])    // Email unique per tenant
}

enum Role {
  ADMIN
  DIRECTOR
  SUPERVISOR
  SENIOR_AUDITOR
  STAFF_AUDITOR
  QA_REVIEWER
  AUDITEE
  VIEWER
}
```

### 1.2.3 — Standards & Configuration Tables

```prisma
model Standard {
  id            String   @id @default(cuid())
  code          String   @unique    // "GAGAS", "IIA_GIAS", "SOX_PCAOB"
  name          String              // "Generally Accepted Government Auditing Standards"
  version       String              // "2024 Revision"
  description   String?
  isActive      Boolean  @default(true)
  config        Json                // Standard-specific configuration
  // Templates, checklists, terminology stored as JSON or related tables
  
  tenants       TenantStandard[]
  requirements  StandardRequirement[]
}

model TenantStandard {
  id            String   @id @default(cuid())
  tenantId      String
  tenant        Tenant   @relation(fields: [tenantId], references: [id])
  standardId    String
  standard      Standard @relation(fields: [standardId], references: [id])
  isDefault     Boolean  @default(false)
  customConfig  Json     @default("{}")  // Tenant overrides
  activatedAt   DateTime @default(now())

  @@unique([tenantId, standardId])
}

model StandardRequirement {
  id            String   @id @default(cuid())
  standardId    String
  standard      Standard @relation(fields: [standardId], references: [id])
  section       String              // "§6.39", "Standard 2100"
  title         String
  description   String
  category      String              // "Independence", "Fieldwork", "Reporting"
  isMandatory   Boolean  @default(true)
  
  crosswalks    Crosswalk[]
}

model Crosswalk {
  id              String   @id @default(cuid())
  requirementId   String
  requirement     StandardRequirement @relation(fields: [requirementId], references: [id])
  mappedStandardId String
  mappedSection    String            // The equivalent section in another standard
  mappingType      String            // "equivalent", "partial", "related"
  notes           String?
}
```

### 1.2.4 — Audit Engine Tables

```prisma
model Engagement {
  id            String   @id @default(cuid())
  tenantId      String
  tenant        Tenant   @relation(fields: [tenantId], references: [id])
  
  title         String
  engagementNumber String?
  type          EngagementType
  status        EngagementStatus @default(PLANNING)
  
  // Standard(s) applicable to this engagement
  standards     EngagementStandard[]
  
  // Scope
  objectives    String?
  scope         String?
  methodology   String?
  
  // Dates
  plannedStartDate DateTime?
  plannedEndDate   DateTime?
  actualStartDate  DateTime?
  actualEndDate    DateTime?
  
  // Risk
  riskRating    RiskRating?
  
  // Relations
  team          EngagementTeam[]
  findings      Finding[]
  workpapers    Workpaper[]
  workPrograms  WorkProgram[]
  approvals     Approval[]
  reports       Report[]
  timeEntries   TimeEntry[]
  
  createdById   String
  createdBy     User   @relation("EngagementCreator", fields: [createdById], references: [id])
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
}

enum EngagementType {
  FINANCIAL_AUDIT
  PERFORMANCE_AUDIT
  COMPLIANCE_AUDIT
  ATTESTATION
  INTERNAL_AUDIT
  IT_AUDIT
  INVESTIGATION
  AGREED_UPON_PROCEDURES
  FOLLOW_UP
  SPECIAL_PROJECT
}

enum EngagementStatus {
  DRAFT
  PLANNING
  FIELDWORK
  REPORTING
  REVIEW
  ISSUED
  CLOSED
  CANCELLED
}

model EngagementStandard {
  id            String     @id @default(cuid())
  engagementId  String
  engagement    Engagement @relation(fields: [engagementId], references: [id])
  standardCode  String     // "GAGAS", "IIA_GIAS"
  isPrimary     Boolean    @default(false)
  
  @@unique([engagementId, standardCode])
}

model Finding {
  id            String   @id @default(cuid())
  tenantId      String
  engagementId  String
  engagement    Engagement @relation(fields: [engagementId], references: [id])
  
  title         String
  findingNumber String?
  status        FindingStatus @default(DRAFT)
  riskRating    RiskRating?
  type          String?       // "Deficiency", "Significant Deficiency", "Material Weakness"
  
  // GAGAS Four Elements (§6.39) — also applicable to other standards
  criteria      String?       // What should be
  condition     String?       // What is
  cause         String?       // Why the gap exists
  effect        String?       // Impact/consequence
  
  // Additional fields
  background    String?
  auditResponse String?
  
  recommendations Recommendation[]
  approvals       Approval[]
  
  createdById   String
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
}

enum FindingStatus {
  DRAFT
  IN_REVIEW
  APPROVED
  PUBLISHED
  CLOSED
}

enum RiskRating {
  CRITICAL
  HIGH
  MEDIUM
  LOW
  INFORMATIONAL
}
```

### 1.2.5 — Supporting Tables

```prisma
model Recommendation {
  id            String   @id @default(cuid())
  tenantId      String
  findingId     String
  finding       Finding  @relation(fields: [findingId], references: [id])
  
  title         String
  description   String?
  priority      Priority @default(MEDIUM)
  status        RecommendationStatus @default(OPEN)
  
  // Management response
  managementResponse String?
  agreedActionPlan   String?
  targetDate         DateTime?
  responsiblePerson  String?
  
  correctiveActions  CorrectiveAction[]
  
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
}

model CorrectiveAction {
  id                String   @id @default(cuid())
  tenantId          String
  recommendationId  String
  recommendation    Recommendation @relation(fields: [recommendationId], references: [id])
  
  title             String
  description       String?
  status            CAPStatus @default(NOT_STARTED)
  dueDate           DateTime?
  completedDate     DateTime?
  evidence          String?
  verifiedById      String?
  verifiedAt        DateTime?
  
  createdAt         DateTime @default(now())
  updatedAt         DateTime @updatedAt
}

model Workpaper {
  id            String   @id @default(cuid())
  tenantId      String
  engagementId  String
  engagement    Engagement @relation(fields: [engagementId], references: [id])
  
  title         String
  reference     String?    // WP reference number
  description   String?
  fileName      String
  fileSize      Int
  mimeType      String
  storagePath   String     // S3 key
  version       Int        @default(1)
  
  status        WorkpaperStatus @default(DRAFT)
  reviewedById  String?
  reviewedAt    DateTime?
  reviewNotes   String?
  
  preparedById  String
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
}

model WorkProgram {
  id            String   @id @default(cuid())
  tenantId      String
  engagementId  String
  engagement    Engagement @relation(fields: [engagementId], references: [id])
  
  title         String
  objective     String?
  scope         String?
  procedures    Json       // Array of procedure steps
  status        String     @default("Not Started")
  
  assignedToId  String?
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
}

model Approval {
  id            String   @id @default(cuid())
  tenantId      String
  
  // Polymorphic: can approve engagements, findings, reports, etc.
  entityType    String   // "engagement", "finding", "report", "planning_doc"
  entityId      String
  
  workflowType  String   // "ENGAGEMENT_APPROVAL", "FINDING_APPROVAL", etc.
  step          Int      // Current step in workflow
  totalSteps    Int
  status        ApprovalStatus @default(PENDING)
  
  requestedById String
  assignedToId  String
  
  comments      String?
  decidedAt     DateTime?
  
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
  
  @@index([entityType, entityId])
}

enum ApprovalStatus {
  PENDING
  APPROVED
  REJECTED
  DELEGATED
  RECALLED
}

model AuditLog {
  id            String   @id @default(cuid())
  tenantId      String
  userId        String
  action        String   // "CREATE", "UPDATE", "DELETE", "APPROVE", "LOGIN"
  entityType    String   // "engagement", "finding", "workpaper"
  entityId      String?
  changes       Json?    // Before/after diff
  ipAddress     String?
  userAgent     String?
  timestamp     DateTime @default(now())
  
  @@index([tenantId, entityType, entityId])
  @@index([tenantId, userId, timestamp])
}

model EngagementTeam {
  id            String   @id @default(cuid())
  engagementId  String
  engagement    Engagement @relation(fields: [engagementId], references: [id])
  userId        String
  role          String   // "Lead Auditor", "Staff Auditor", "Supervisor", "QA Reviewer"
  assignedAt    DateTime @default(now())
  
  @@unique([engagementId, userId])
}

model TimeEntry {
  id            String   @id @default(cuid())
  tenantId      String
  engagementId  String?
  engagement    Engagement? @relation(fields: [engagementId], references: [id])
  userId        String
  
  date          DateTime
  hours         Float
  description   String?
  category      String?  // "Planning", "Fieldwork", "Reporting", "Admin", "CPE"
  status        String   @default("Draft") // "Draft", "Submitted", "Approved"
  
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
}

model Report {
  id            String   @id @default(cuid())
  tenantId      String
  engagementId  String
  engagement    Engagement @relation(fields: [engagementId], references: [id])
  
  title         String
  type          String   // "Draft", "Final", "Summary"
  status        String   @default("Draft")
  version       Int      @default(1)
  
  // Content sections (standard-dependent)
  content       Json     // Flexible content structure per standard
  
  // PDF
  pdfStoragePath String?
  
  createdById   String
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
}
```

## Notes

- All IDs use `cuid()` (collision-resistant, URL-safe, sortable)
- All tenant-scoped tables include `tenantId` for RLS
- `Json` fields used for flexible, standard-dependent configuration
- Soft delete not implemented initially — use `AuditLog` for recovery if needed
- Consider partitioning `AuditLog` by month for large deployments
